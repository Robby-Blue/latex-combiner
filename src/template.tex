\newcommand{\makecustomtitlepage}{
  \begin{titlepage}
    \begin{center}
      \vspace*{1cm}
      {\$title\$ \par}
      \vspace{0.2cm}
      {\$subtitle\$ \par}
      \vspace{2.3cm}
      {\$author\$ \par}
      \vspace{2.3cm}
      {\$explanation\$ \par}
      \vspace{4cm}
      \vfill
      {\$date\$}
    \end{center}
  \end{titlepage}
}

  \section{#1 \label{#1}}
}

}

}

\newcommand{\documentsubsubsubsection}[1]{

}

  \vspace*{1\baselineskip} \par

}
